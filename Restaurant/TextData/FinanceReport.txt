CREATE PROCEDURE Report_FinancialResultTotals 
	(@WeekNumber int, @Year int)
AS	
BEGIN
	declare @StartYear datetime2 = cast(cast(@Year as char(4))+'-01-01' as date);
	declare @StartDate datetime2 = dbo.GetFirstDayOfWeek(@WeekNumber, @Year);
	declare @FinishDate datetime2 = dateadd(day, 7, dateadd(millisecond, -1, @StartDate));
	
	declare @StartDebt numeric(25,2) = isnull((select Sum([Sum]) from [dbo].[GetDebtsBalance] (@StartDate, -1)), 0);
	declare @IncomeDebt numeric(25,2) = isnull((select Sum([Sum]) from Remainders_DebtsBalance_History where [Sum]&gt;0 and MotionDate between @StartDate and @FinishDate), 0);
	declare @OutcomeDebt numeric(25,2) = isnull((select Sum(-[Sum]) from Remainders_DebtsBalance_History where [Sum]&lt;0 and MotionDate between @StartDate and @FinishDate), 0);
	
	declare @UtkId bigint = isnull((select cast(Value as bigint) from SystemConsts where Description = 'UtkContractor'), 0);
	declare @BonusesArticle bigint = isnull((select cast(Value as bigint) from SystemConsts where Description = 'BonusesReturning'), 0);
	declare @ArticleB bigint = isnull((select cast(Value as bigint) from SystemConsts where Description = 'ArticleB'), 0);
	declare @ArticleP bigint = isnull((select cast(Value as bigint) from SystemConsts where Description = 'ArticleP'), 0);
	
	declare @PeriodStartDate datetime2 = case when @StartYear&lt;@StartDate then @StartYear else @StartDate end;
	
	declare @WeekExpensesCash numeric(25,2), @TotalExpenses numeric(25,2), 
		@WeekSoldNonCash numeric(25,2), @WeekSoldCash numeric(25,2), @WeekBonus numeric(25,2),
		@BonusesRemainders numeric(25,2),
		@ProductIncomeFromUTKSum numeric(25,2);
	
	declare @paidBonusesToUtkPerYear decimal(25,2), @paidB decimal(25,2), @paidP decimal(25,2);
	
	select @WeekExpensesCash = Sum(case when [Date] between @StartDate and @FinishDate and Cash = 1 then [Sum] else 0 end),
		@TotalExpenses =  Sum(case when [Date] between @StartYear and @FinishDate and Cash = 1 then [Sum] else 0 end),
		@paidBonusesToUtkPerYear = Sum(case when [Date] between @StartYear and @FinishDate 
			and Contractor = @UtkId
			and Article = @BonusesArticle
			then [Sum] else 0 end),
		@paidB = Sum(case when [Date] between @StartYear and @FinishDate and Article = @ArticleB then [Sum] else 0 end),
		@paidP = Sum(case when [Date] between @StartYear and @FinishDate and Article = @ArticleP then [Sum] else 0 end)
		from Expenses where [Date] between @PeriodStartDate and @FinishDate and MarkForDeleting = 0;
	
	declare @IncomePerWeek decimal(25,2);
	declare @IncomePerWeekCash decimal(25,2);
	
	select @WeekSoldCash = Sum(case when [Date] between @StartDate and @FinishDate and Cash = 1 then [Sum] else 0 end),
		@WeekSoldNonCash =  Sum(case when [Date] between @StartDate and @FinishDate and Cash = 0 then [Sum] else 0 end),
		@IncomePerWeekCash = Sum(case when [Date] between @StartDate and @FinishDate and Cash = 1 then Paid else 0 end),
		@IncomePerWeek = Sum(case when [Date] between @StartDate and @FinishDate and Cash = 0 then Paid else 0 end),
		@WeekBonus =  Sum(case when [Date] between @StartDate and @FinishDate then [Bonus] else 0 end)
		from Shipment where [Date] between @PeriodStartDate and @FinishDate and MarkForDeleting = 0;
	
	select @ProductIncomeFromUTKSum = Sum(case when [Date] between @StartYear and @FinishDate then [Sum] else 0 end)
		from Acception where [Date] between @PeriodStartDate and @FinishDate and MarkForDeleting = 0;
		
	declare @bodyLoanRemainder as decimal(25,2);
	declare @percentsLoanRemainder as decimal(25,2);
	
	with loans as (
select cap.Id, cap.BodyRemainder, cap.[Percent] P, cap.Date, isnull(max(s.LineNumber), -1) LastLine
	from Loan cap 
	left join SubLoanRepayment s on s.IdDoc = cap.Id and cap.Active = 1 and cap.MarkForDeleting = 0
	group by cap.Id, cap.BodyRemainder, cap.[Percent], cap.Date
)
select @bodyLoanRemainder = Sum(cap.BodyRemainder), 
	@percentsLoanRemainder = Sum(isnull(s.PercentsReminder,0) + cap.BodyRemainder * cap.P 
		* DATEDIFF(day, cast(isnull(Expenses.Date, cap.Date) as date), cast(@FinishDate as date)) / 36500)
	from loans cap
	left join SubLoanRepayment s on cap.Id = s.IdDoc and s.LineNumber = cap.LastLine
	left join Expenses on Expenses.Id = s.Expenses;
	
	declare @DebitsCash decimal(25,2) = isnull((select Sum(Summa) from dbo.[GetDebitsBalance](@FinishDate, 0, 1)), 0);
	declare @DebitsNonCash decimal(25,2) = isnull((select Sum(Summa) from dbo.[GetDebitsBalance](@FinishDate, 0, 0)), 0);
	
	declare @CashStartBalance decimal(25,2) = isnull((select Sum(Summa) from dbo.[GetCashFlow](@StartDate, 1)), 0);
	declare @NonCashStartBalance decimal(25,2) = isnull((select Sum(Summa) from dbo.[GetCashFlow](@StartDate, 0)), 0);
	 
	declare @CashFinishBalance decimal(25,2) = isnull((select Sum(Summa) from dbo.[GetCashFlow](@FinishDate, 1)), 0);
	declare @NonCashFinishBalance decimal(25,2) = isnull((select Sum(Summa) from dbo.[GetCashFlow](@FinishDate, 0)), 0);
	
	declare @CashOutcome decimal(25,2);
	declare @NonCashOutcome decimal (25,2);
	
	select @CashOutcome = ISNULL(-Sum(case when h.Cash = 1 then h.Summa else 0 end), 0),
		@NonCashOutcome = ISNULL(-Sum(case when h.Cash = 0 then h.Summa else 0 end), 0)
		from dbo.Remainders_CashFlow_History h
		where h.MotionDate between @StartDate and @FinishDate and h.Summa &lt; 0
	
	declare @NonCashOnUTK decimal(25,2), @PaidToUtkPerYear decimal(25,2);
	
	select @NonCashOnUTK = isnull(Sum(case when cap.Date &gt;= @StartDate and cap.Cash = 0 then cap.Sum else 0 end), 0),
		@PaidToUtkPerYear = isnull(Sum(case when cap.Date &gt;= @StartYear then cap.Sum else 0 end), 0)
	from MoneyOutcome cap
	where cap.date between @PeriodStartDate and @FinishDate and cap.Contractor = @UtkId 
		and cap.MarkForDeleting = 0 
	
	declare @DebtsToUTK decimal(25,2);
	declare @DebtsToAnotherSuppliers decimal(25,2);
	
	select @DebtsToUTK = isnull(sum(case when b.Contractor = @UtkId then b.Summa else 0 end),0),
		@DebtsToAnotherSuppliers = isnull(sum(case when b.Contractor = @UtkId then 0 else b.Summa end),0)
		from dbo.GetDebtsToSuppliers(@FinishDate, 0, 0) b;
		
	select @BonusesRemainders = isnull(sum(Summa),0) from dbo.GetBonusesBalance(@FinishDate, 0);
	
	declare @CashToCashBox decimal(25,2) = isnull((select Sum(c.[Sum]) from MoneyIncome c where c.Date between @StartDate and @FinishDate and c.MarkForDeleting = 0 and c.CashToСashbox = 1), 0);
	declare @kdd decimal(25,2) = (@NonCashFinishBalance + @CashFinishBalance) + (@DebitsNonCash + @DebitsCash) - @DebtsToUTK - @DebtsToAnotherSuppliers;
    
    declare @productionSum decimal(25,2);
    
    -- getting sum remainder BEGIN
    declare @addDaysToDate int = (
	case 
		when @WeekNumber &lt; 5 
			then 50 
		when @WeekNumber &gt; 50 
			then -50 
		else 
			0

CREATE PROCEDURE Report_FinancialResultExpenses
	(@WeekNumber int, @Year int)
AS	
BEGIN
	declare @StartYear datetime2 = cast(cast(@Year as char(4))+'-01-01' as date);
	
	declare @StartDate datetime2 = dbo.GetFirstDayOfWeek(@WeekNumber, @Year);
	declare @FinishDate datetime2 = dateadd(day, 7, dateadd(millisecond, -1, @StartDate));
	
	declare @PeriodStartDate datetime2 = case when @StartYear&lt;@StartDate then @StartYear else @StartDate end;
	
	select Sum(case when e.[Date] between @StartDate and @FinishDate then e.[Sum] else 0 end) WeekExpenses,
		Sum(case when e.[Date] between @StartYear and @FinishDate then e.[Sum] else 0 end) YearExpenses,
		Cash, isnull(RTRIM(a.Description),'') Article, isnull(RTRIM(et.Description),'') ExpensesType,
		case when cash = 1 then 'Нал' else 'Б/Н' end IsCashName
		from Expenses e
		left join ExpensesArticles a on a.Id = e.Article
		left join ExpensesTypes et on et.Id = e.ExpensesType
		where e.[Date] between @PeriodStartDate and @FinishDate and e.MarkForDeleting = 0
		group by Cash, a.Description, et.Description
END

	(@WeekNumber int, @Year int)
AS

declare @addDaysToDate int = (
	case

			then 50

select * from dbo.FinanceResult(@WeekNumber, @Year, @addDaysToDate, @AveragePricesLastWeekId, @AveragePricesCurrentWeekId);
